    常用命令

    Linux命令大全(手册)：https://www.linuxcool.com/

    目录

    # 新建目录
mkdir 目录名
    # 无法创建目录"xxxx": 权限不够
root用户登录或者chmod
chmod +x ./*         //为当前文件夹下所有文件赋予可执行权限
chmod 777 install 
    # 删除空目录
rmdir 目录名

    递归地删除目录，使用递归选项-r或-R，如果目录是受写保护的，则会提示是否继续删除目录和目录中的文件作为一个整体。（慎重）

    # 删除非空目录
rm -rf 目录名/      //-f 强制删除而不被提示
# 单个命令中删除多个目录，如下面的命令所示。命令删除所有目录和它们的子目录，而不提示删除 
rm -rf dir1 dir2 dir3

    进程

    # 查看所有进程
ps -ef
    -e：显示系统内所有进程的信息。与 -A 选项功能相同
    -f：使用完整 (full) 的格式显示进程信息，如果只有 ps -e 则输出进程信息的格式和只使用 ps 一样
# 根据特定情况过滤
ps -ef | grep mmp-front  //mmp-front是进程关键字 grep 是搜索
ps -aux | grep java      //-aux 显示所有状态

kill -9 [PID]            //-9 表示强迫进程立即停止

    防火墙

    CentOS7 的防火墙配置
# 查看防火墙状态
systemctl status firewalld
# 关闭防火墙：
systemctl stop firewalld.service
# 开启防火墙：
systemctl start firewalld.service
# 关闭开机启动：
systemctl disable firewalld.service
# 开启开机启动：
systemctl enable firewalld.service

    # 解压安装包
tar -zxvf xxx.tar.gzbash

vim /etc/profile
    # 添加环境变量（profile）尾部加⼊ xx 的 bin 路径配置
export xx_HOME=/usr/local/xx
export PATH=$PATH:$xx_HOME/bin
    # 使环境变量⽣效
source /etc/profile
