    在ip:port前面加上@用来表示密码，比如password@ip:port ，现在才知道能看到源码的程序真的很幸福。

    private void Showdata_Click(object sender, EventArgs e)//160.0.3.5  ArrayList<byte> = get(20122)
        {
            MemoryStream memx=null;
            BinaryWriter binwritex=null;
            BinaryReader binreadx=null;

            try {
                byte[] data = new byte[4096];
                 RedisClient client = new RedisClient("160.0.3.5", 6379,"12345",0);

                data = client.Get<byte[]>(this.lbTelName.Text);

                if (data != null)
                {

                    memx = new MemoryStream();
                    binwritex = new BinaryWriter(memx);
                    binreadx = new BinaryReader(memx);
                    binwritex.Write(data);
                    for (int i = 0; i < this.dgvTelegram.RowCount - 1; i++)
                    {
                        short offset = Convert.ToInt16(dgvTelegram.Rows[i].Cells["OFFSET"].Value);
                        short length = Convert.ToInt16(dgvTelegram.Rows[i].Cells["LENGTH"].Value);
                        string type = Convert.ToString(dgvTelegram.Rows[i].Cells["TYPE"].Value);

                        if (offset + length < data.Length)
                        {
                            memx.Position = offset;
                            if (type == "DINT" || type == "DWORD")            //int
                            {
                                dgvTelegram.Rows[i].Cells["SNAPSHOOT"].Value = binreadx.ReadInt32();
                            }
                            else if (type == "FLOAT")            //float 
                            {
                                dgvTelegram.Rows[i].Cells["SNAPSHOOT"].Value = binreadx.ReadSingle();
                            }
                            else if (type == "INT" || type == "WORD")             //short
                            {
                                dgvTelegram.Rows[i].Cells["SNAPSHOOT"].Value = binreadx.ReadInt16();
                            }
                            else if (type == "BOOL")             //short
                            {
                                dgvTelegram.Rows[i].Cells["SNAPSHOOT"].Value = binreadx.ReadInt16();
                            }
                            else if (type == "string" || type == "String")             //short
                            {
                                string str2 = binreadx.ReadString();
                                str2=str2.Substring(0,str2.Length);
                                dgvTelegram.Rows[i].Cells["SNAPSHOOT"].Value = str2;

                            }
                            else
                            {
                                MessageBox.Show("row:" + i + " type execption "+ type);
                            }
                        }
                    }

                }
                else
                {
                    MessageBox.Show("telegramId exist");
                }
            }
            catch (System.Exception ex)
            {
                MessageBox.Show(ex.Message);
            }
            finally
            {
                if (binwritex != null)
                    binwritex.Close();
                if (binreadx != null)
                    binreadx.Close();
                if (memx != null)
                    memx.Close();
            }

        }
